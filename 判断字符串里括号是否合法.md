```javascript
Main(string){
var stack =new Stack();
    for(var i =0;i<string.length;i++){
     var item = string [i];
        if(item=="("){
            stack.push(item);
        }else if(item==")"){
            if(stack.isEmpty()){
                return false
            }else{
                stack.pop();
            }
        }
    }
    return stack.isEmpty();
};
```
